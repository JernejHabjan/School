ABSOLUTE, RELATIVE, FIXED POSITIONING:
	Absolute positioning:
		#apples{border:2px solid red;
		position:absolute; //ne skrbi a je kšn element že na tej lokacij... pač ga prekrije
			//it makes problems of overlapping
		width:150px;
		height:175px;
		top:30px;
		left:15px;
		}	
		<div id="apples">something</div>
	Relative positioning:
		#apples{border:2px solid red;
			position:relative; //ne premakne ga od levo gor 30px dol in 15 desno ampak od kjer bi bil po defaultu
				//it makes problems of gaps
			width:150px;
			height:175px;
			top:30px;
			left:15px;
			}	
		<p id="apples">something</p>
	Fixed positioning:
		#apples{border:2px solid red;
			position:fixed; //TUD ČE SCROLLAŠ OSTANE NA ISTMU MESTU

			height:175px;
			top:30px;
			left:15px;
			}	
		<p id="apples">something</p>
